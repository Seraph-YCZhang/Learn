class Solution(object):
    def maxTurbulenceSize(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A:
            return 0
        res = 0 
        start, end = 0, 1
        l = len(A)
        if l < 2:
            return 1
        if A.count(A[0]) == l:
            return 1
        while start < l - 2 and end < l - 1:
            if (A[end] < A[end + 1] and A[end] < A[end - 1]) or (A[end] > A[end + 1] and A[end] > A[end - 1]) :
                end += 1
            else:
                print(A[start:end])
                res = max(res, end - start + 1)
                start, end = end, end + 1
        res = max(res, end - start + 1)
        return res
